**Section 1: Species table** (per CAM-03)

```sql
CREATE TABLE IF NOT EXISTS public.species (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name        text NOT NULL UNIQUE,
  common_name text NOT NULL,
  scientific_name text,
  category    text NOT NULL CHECK (category IN ('mammal', 'bird', 'reptile', 'amphibian', 'insect', 'other')),
  created_at  timestamptz DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_species_name ON public.species(name);
CREATE INDEX IF NOT EXISTS idx_species_category ON public.species(category);
```

Species is a global registry (not org-scoped). RLS: authenticated users can read, service_role can insert/update.

```sql
ALTER TABLE public.species ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Authenticated read species"
  ON public.species FOR SELECT
  USING (auth.role() = 'authenticated');

CREATE POLICY "Service role manage species"
  ON public.species FOR ALL
  USING (auth.role() = 'service_role');
```

**Section 3: camera_events table** (per CAM-01, CAM-06)

```sql
CREATE TABLE IF NOT EXISTS public.camera_events (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  unit_id         text NOT NULL REFERENCES public.units(id),
  org_id          uuid NOT NULL REFERENCES public.organisations(id) ON DELETE CASCADE,
  captured_at     timestamptz NOT NULL,
  image_path      text,
  image_width     int,
  image_height    int,
  model_name      text,
  inference_time_ms int,
  battery_percent int,
  communication_type text,
  detection_count int NOT NULL DEFAULT 0,
  mqtt_topic      text,
  created_at      timestamptz DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_camera_events_org_captured
  ON public.camera_events (org_id, captured_at DESC);
CREATE INDEX IF NOT EXISTS idx_camera_events_unit
  ON public.camera_events (unit_id);
```

RLS policies for camera_events (per CAM-04):

```sql
ALTER TABLE public.camera_events ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Members can view org camera events"
  ON public.camera_events FOR SELECT
  USING (public.trap_can_view_org(org_id));

CREATE POLICY "Service role can insert camera events"
  ON public.camera_events FOR INSERT
  WITH CHECK (auth.role() = 'service_role');

CREATE POLICY "Service role can update camera events"
  ON public.camera_events FOR UPDATE
  USING (auth.role() = 'service_role');

CREATE POLICY "Admins can delete camera events"
  ON public.camera_events FOR DELETE
  USING (public.trap_can_admin_org(org_id));
```

**Section 4: camera_detections table** (per CAM-02)

Bounding box coordinates are normalised 0-1 floats matching NE301 output format.

```sql
CREATE TABLE IF NOT EXISTS public.camera_detections (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  camera_event_id uuid NOT NULL REFERENCES public.camera_events(id) ON DELETE CASCADE,
  species_id      uuid REFERENCES public.species(id),
  class_name      text NOT NULL,
  confidence      real NOT NULL CHECK (confidence >= 0 AND confidence <= 1),
  x               real NOT NULL CHECK (x >= 0 AND x <= 1),
  y               real NOT NULL CHECK (y >= 0 AND y <= 1),
  width           real NOT NULL CHECK (width >= 0 AND width <= 1),
  height          real NOT NULL CHECK (height >= 0 AND height <= 1),
  created_at      timestamptz DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_camera_detections_event
  ON public.camera_detections (camera_event_id);
CREATE INDEX IF NOT EXISTS idx_camera_detections_species
  ON public.camera_detections (species_id);
CREATE INDEX IF NOT EXISTS idx_camera_detections_confidence
  ON public.camera_detections (confidence DESC);
```

RLS for camera_detections (inherits org scope via camera_events join):

```sql
ALTER TABLE public.camera_detections ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Members can view org camera detections"
  ON public.camera_detections FOR SELECT
  USING (
    EXISTS (
      SELECT 1 FROM public.camera_events ce
      WHERE ce.id = camera_detections.camera_event_id
        AND public.trap_can_view_org(ce.org_id)
    )
  );

CREATE POLICY "Service role can insert camera detections"
  ON public.camera_detections FOR INSERT
  WITH CHECK (auth.role() = 'service_role');

CREATE POLICY "Service role can delete camera detections"
  ON public.camera_detections FOR DELETE
  USING (auth.role() = 'service_role');
```
