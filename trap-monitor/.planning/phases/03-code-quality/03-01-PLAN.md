  model.toUpperCase();

  if (model == "EG800Q")
    return new EG800Q();
  else if (model == "SIM7080G")
    return new SIM7080G();
  else if (model == "REDCAP5G")
    return new RedCap5G();
  else
  {

    return new EG800Q();
  }
}
